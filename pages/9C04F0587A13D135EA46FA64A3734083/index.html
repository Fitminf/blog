# MacOS docker 环境搭建

# mysql

  1. docker mysql 环境准备
# 2024-07-05 mysql last version ：9.0
docker pull mysql  

# 记得修改密码 MYSQL_ROOT_PASSWORD=xxxx
docker run -di --name mysql -e MYSQL_ROOT_PASSWORD=xxxx -p 3306:3306 mysql

  1. datagrip 连接 mysql
  2. 创建用户
-- 查看用户
USE mysql;
SELECT * FROM user;

-- 创建内网连接账户 记得修改密码 IDENTIFIED BY 'xxxx'
CREATE USER 'caramel'@'localhost' IDENTIFIED BY 'xxxx';
-- 刷新
FLUSH PRIVILEGES;

-- 创建外网连接账户 记得修改密码 IDENTIFIED BY 'xxxx'
CREATE USER 'caramel'@'%' IDENTIFIED BY 'xxxx';
FLUSH PRIVILEGES;

-- 授权
GRANT ALL PRIVILEGES ON *.* TO 'caramel'@'%'
FLUSH PRIVILEGES;

-- 查看权限
SHOW GRANTS FOR 'caramel'@'%';

docker run -di --name redis -v # /xxx/redis.conf:/usr/local/etc/redis/redis.conf -p 6379:6379 redis redis-server /usr/local/etc/redis/redis.conf

     -v /Users/fitminf/dev/data/minio/data:/data \
