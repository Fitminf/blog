# mysql

docker pull mysql

docker run -di --name=fitminf_mysql -p 3306:3306 -e MYSQL_ROOT_PASSWORD=root -e MYSQL_USER=fitminf -e MYSQL_PASSWORD=fitminf mysql

# minio

docker pull minio/minio

     --name=fitminf_minio \
     -e "MINIO_ACCESS_KEY=fitminf" \
     -e "MINIO_SECRET_KEY=12345678" \
     -v /Users/fitminf/dev/data/minio/data:/data \
     -v /Users/fitminf/dev/data/minio/config:/root/.minio \
     minio/minio server \
     /data --console-address ":9090" -address ":9000"
